						function getAverageCluster(allChildrenMarkers)
						{
							var numberPattern = /\d+/g;
							// need to parse _popup._content to get mic value
							// if we don't want to ajax again.
							var totalMic = 0;
							var avgProbMic = 0;
							var stringInputLeafletData;
							for (var i = 0; i < allChildrenMarkers.length; i++)
							{
								stringInputLeafletData = allChildrenMarkers[i]._popup._content;
								totalMic += parseInt(stringInputLeafletData.match(numberPattern)[1]);
							}
							avgProbMic = totalMic / allChildrenMarkers.length;
							return avgProbMic.toFixed(2);
						}
